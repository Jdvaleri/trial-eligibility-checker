    <script>

        // Simple user database (for demo purposes only)

        const users = {

            "crc1": "password123",

            "crc2": "securepass456"

        };


        // Login functionality

        document.getElementById('loginForm').addEventListener('submit', function(e) {

            e.preventDefault();

            const username = document.getElementById('username').value;

            const password = document.getElementById('password').value;

            

            if (users[username] && users[username] === password) {

                document.getElementById('loginContainer').classList.add('hidden');

                document.getElementById('mainContainer').classList.remove('hidden');

                document.getElementById('loginError').style.display = 'none';

            } else {

                document.getElementById('loginError').style.display = 'block';

            }

        });


        // Logout functionality

        document.getElementById('logoutBtn').addEventListener('click', function() {

            document.getElementById('mainContainer').classList.add('hidden');

            document.getElementById('loginContainer').classList.remove('hidden');

            document.getElementById('loginForm').reset();

        });


        // Eligibility checker

        document.getElementById('patientForm').addEventListener('submit', function(e) {

            e.preventDefault();

            

            // Get form values

            const age = parseInt(document.getElementById('age').value);

            const bcdva = parseFloat(document.getElementById('bcdva').value);

            const geoAtrophy = document.getElementById('geoAtrophy').checked;

            const disciformScar = document.getElementById('disciformScar').checked;

            const noCNV = document.getElementById('noCNV').checked;

            const cataract = document.getElementById('cataract').value;

            const telescopeTraining = parseFloat(document.getElementById('telescopeTraining').value);

            const peripheralVision = document.getElementById('peripheralVision').value;

            const cognitiveImpairment = document.getElementById('cognitiveImpairment').checked;

            const stargardt = document.getElementById('stargardt').checked;

            const diabeticRetinopathy = document.getElementById('diabeticRetinopathy').checked;

            const iop = parseInt(document.getElementById('iop').value);

            const acd = parseFloat(document.getElementById('acd').value);

            const ecd = parseInt(document.getElementById('ecd').value);


            let isEligible = true;

            let reasons = [];


            // Inclusion Criteria

            if (age < 65) {

                isEligible = false;

                reasons.push("Age must be at least 65 years");

            }


            if (bcdva < 0.9 || bcdva > 1.6) {

                isEligible = false;

                reasons.push("ETDRS BCDVA must be between 0.9 and 1.6 logMAR");

            }


            if (!geoAtrophy && !disciformScar) {

                isEligible = false;

                reasons.push("Must have geographic atrophy or disciform scar in both eyes");

            }


            if (!noCNV) {

                isEligible = false;

                reasons.push("Must have no active CNV in past 6 months");

            }


            if (cataract !== "yes") {

                isEligible = false;

                reasons.push("Implanted eye must have visually significant cataract");

            }


            if (telescopeTraining < 0.1) {

                isEligible = false;

                reasons.push("Must achieve at least 0.1 logMAR improvement with telescope training");

            }


            if (peripheralVision !== "yes") {

                isEligible = false;

                reasons.push("Non-implanted eye must have adequate peripheral vision");

            }


            // Exclusion Criteria

            if (cognitiveImpairment) {

                isEligible = false;

                reasons.push("Cognitive impairment present");

            }


            if (stargardt) {

                isEligible = false;

                reasons.push("Stargardt macular dystrophy present");

            }


            if (diabeticRetinopathy) {

                isEligible = false;

                reasons.push("Diabetic retinopathy present");

            }


            if (iop > 22) {

                isEligible = false;

                reasons.push("IOP must not exceed 22 mmHg");

            }


            if (acd < 3.0) {

                isEligible = false;

                reasons.push("Anterior chamber depth must be at least 3.0 mm");

            }


            if (age >= 65 && age <= 84 && ecd < 2000) {

                isEligible = false;

                reasons.push("ECD must be at least 2000 cells/mm² for ages 65-84");

            } else if (age >= 85 && ecd < 1800) {

                isEligible = false;

                reasons.push("ECD must be at least 1800 cells/mm² for ages 85+");

            }


            // Note: Some criteria (e.g., history of surgery, myopia/hyperopia) require additional fields if needed


            // Display result

            const resultDiv = document.getElementById('result');

            resultDiv.style.display = 'block';

            

            if (isEligible) {

                resultDiv.className = 'eligible';

                resultDiv.innerHTML = '<h3>Patient is Eligible</h3><p>Patient meets all inclusion criteria and no exclusion criteria.</p>';

            } else {

                resultDiv.className = 'not-eligible';

                resultDiv.innerHTML = '<h3>Patient is Not Eligible</h3><p>Reasons:</p><ul>' + 

                    reasons.map(reason => `<li>${reason}</li>`).join('') + '</ul>';

            }

        });
